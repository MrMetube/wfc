%     % \caption{
%     %     Formel für die Shannon-Entropie. 
%     %     $\mathcal{X}$ ist die Menge der Möglichen Zustände einer Zelle.
%     %     $p(x)$ ist die Wahrscheinlichkeit das $x$ gewählt wird.
%     % }

\begin{equation}
    \label{eq:entropy}
    \mathrm{entropy}(\mathcal{X}) = -\sum_{x \in \mathcal{X}} p(x),\log p(x)
\end{equation}
